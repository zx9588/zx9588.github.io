  1. SIMPLE

    жңҖз®ҖеҚ•зҡ„select(дёҚдҪҝз”ЁUNIONжҲ–еӯҗжҹҘиҜўзӯү) дҫӢеҰӮ:

    explain select * from t3 where id=123;

  2. PRIMARY

    жңҖеӨ–еұӮзҡ„selectпјҢдё»жҹҘиҜўпјҢдҫӢеҰӮпјҡ

    explain select * from (select * from t3 where id=123) a ;

  3. UNION

    UNIONдёӯзҡ„з¬¬дәҢдёӘжҲ–еҗҺйқўзҡ„selectиҜӯеҸҘпјҢдҫӢеҰӮпјҡ

    explain select * from t3 where id=123 union all select * from t3 ;

  4. DEPENDENT UNION

    UNIONдёӯзҡ„з¬¬дәҢдёӘжҲ–еҗҺйқўзҡ„selectиҜӯеҸҘпјҢеҸ–еҶідәҺеӨ–йқўзҡ„жҹҘиҜўпјҢдҫӢеҰӮпјҡ

    explain select * from t3 where id in (select id from t3 where id=123 union all select id from t3) ;

  5. UNION RESULT

    UNIONзҡ„з»“жһңпјҢunionиҜӯеҸҘдёӯз¬¬дәҢдёӘselectејҖе§ӢеҗҺйқўжүҖжңүselect

  6. SUBQUERY

    йқһжңҖеӨ–еұӮзҡ„selectпјҢеңЁеӯҗжҹҘиҜўдёӯз¬¬дёҖдёӘselectпјҢдҫӢеҰӮпјҡ

    explain select * from t3 where id = (select id from t3 where id=123 ) ;

  7. DEPENDENT SUBQUERY

    еӯҗжҹҘиҜўдёӯзҡ„з¬¬дёҖдёӘselectпјҢеҸ–еҶідәҺеӨ–йқўзҡ„жҹҘиҜўпјҢдҫӢеҰӮпјҡ

    explain select id from t3 where id in (select id from t3 where id=123 ) ;

  8. DERIVED

    жҙҫз”ҹиЎЁзҡ„select(fromеӯҗеҸҘзҡ„еӯҗжҹҘиҜў)

    explain select * from (select * from t3 where id=123) a ;
