  • ж ‘жһқиҠӮзӮ№пјҲCompositeпјүпјҡе®ҡд№үж ‘жһқиҠӮзӮ№зҡ„иЎҢдёәпјҢеӯҳеӮЁеӯҗиҠӮзӮ№пјҢз»„еҗҲж ‘жһқиҠӮзӮ№е’ҢеҸ¶еӯҗиҠӮзӮ№еҪўжҲҗдёҖдёӘж ‘еҪўз»“жһ„гҖӮ
  • еҸ¶еӯҗиҠӮзӮ№пјҲLeafпјүпјҡеҸ¶еӯҗиҠӮзӮ№еҜ№иұЎпјҢе…¶дёӢеҶҚж— еҲҶж”ҜпјҢжҳҜзі»з»ҹеұӮж¬ЎйҒҚеҺҶзҡ„жңҖе°ҸеҚ•дҪҚгҖӮ

жЎҲдҫӢе®һзҺ°

гҖҗдҫӢгҖ‘иҪҜд»¶иҸңеҚ•

еҰӮдёӢеӣҫпјҢжҲ‘д»¬еңЁи®ҝй—®еҲ«зҡ„дёҖдәӣз®ЎзҗҶзі»з»ҹж—¶пјҢз»ҸеёёеҸҜд»ҘзңӢеҲ°зұ»дјјзҡ„иҸңеҚ•гҖӮдёҖдёӘиҸңеҚ•еҸҜд»ҘеҢ…еҗ«иҸңеҚ•йЎ№пјҲиҸңеҚ•йЎ№жҳҜжҢҮдёҚеҶҚеҢ…еҗ«е…¶д»–еҶ…е®№зҡ„иҸңеҚ•жқЎзӣ®пјүпјҢд№ҹеҸҜд»ҘеҢ…еҗ«еёҰжңүе…¶д»–иҸңеҚ•йЎ№зҡ„иҸңеҚ•пјҢеӣ жӯӨдҪҝз”Ёз»„еҗҲжЁЎејҸжҸҸиҝ°иҸңеҚ•е°ұеҫҲжҒ°еҪ“пјҢжҲ‘д»¬зҡ„йңҖжұӮжҳҜй’ҲеҜ№дёҖдёӘиҸңеҚ•пјҢжү“еҚ°еҮәе…¶еҢ…еҗ«зҡ„жүҖжңүиҸңеҚ•д»ҘеҸҠиҸңеҚ•йЎ№зҡ„еҗҚз§°гҖӮ

иҰҒе®һзҺ°иҜҘжЎҲдҫӢпјҢжҲ‘д»¬е…Ҳз”»еҮәзұ»еӣҫпјҡ

д»Јз Ғе®һзҺ°пјҡ

дёҚз®ЎжҳҜиҸңеҚ•иҝҳжҳҜиҸңеҚ•йЎ№пјҢйғҪеә”иҜҘз»§жүҝиҮӘз»ҹдёҖзҡ„жҺҘеҸЈпјҢиҝҷйҮҢе§‘дё”е°ҶиҝҷдёӘз»ҹдёҖзҡ„жҺҘеҸЈз§°дёәиҸңеҚ•з»„д»¶гҖӮ

//иҸңеҚ•з»„д»¶  дёҚз®ЎжҳҜиҸңеҚ•иҝҳжҳҜиҸңеҚ•йЎ№пјҢйғҪеә”иҜҘз»§жүҝиҜҘзұ»
public abstract class MenuComponent {
    protected String name;
    protected int level;

    //ж·»еҠ иҸңеҚ•
    public void add(MenuComponent menuComponent){
        throw new UnsupportedOperationException();
    }

    //з§»йҷӨиҸңеҚ•
    public void remove(MenuComponent menuComponent){
        throw new UnsupportedOperationException();
    }

    //иҺ·еҸ–жҢҮе®ҡзҡ„еӯҗиҸңеҚ•
    public MenuComponent getChild(int i){
        throw new UnsupportedOperationException();
    }

    //иҺ·еҸ–иҸңеҚ•еҗҚз§°
    public String getName(){
        return name;
    }

    public void print(){
        throw new UnsupportedOperationException();
    }
}

иҝҷйҮҢзҡ„MenuComponentе®ҡд№үдёәжҠҪиұЎзұ»пјҢеӣ дёәжңүдёҖдәӣе…ұжңүзҡ„еұһжҖ§е’ҢиЎҢдёәиҰҒеңЁиҜҘзұ»дёӯе®һзҺ°пјҢMenuе’ҢMenuItemзұ»е°ұеҸҜд»ҘеҸӘиҰҶзӣ–иҮӘе·ұж„ҹе…ҙи¶Јзҡ„ж–№жі•пјҢиҖҢдёҚз”ЁжҗӯзҗҶдёҚйңҖиҰҒжҲ–иҖ…дёҚж„ҹе…ҙи¶Јзҡ„ж–№жі•пјҢдёҫдҫӢжқҘиҜҙпјҢMenuзұ»еҸҜд»ҘеҢ…еҗ«еӯҗиҸңеҚ•пјҢеӣ жӯӨйңҖиҰҒиҰҶзӣ–add()гҖҒremove()гҖҒgetChild()ж–№жі•пјҢдҪҶжҳҜMenuItemе°ұдёҚеә”иҜҘжңүиҝҷдәӣж–№жі•гҖӮиҝҷйҮҢз»ҷеҮәзҡ„й»ҳи®Өе®һзҺ°жҳҜжҠӣеҮәејӮеёёпјҢдҪ д№ҹеҸҜд»Ҙж №жҚ®иҮӘе·ұзҡ„йңҖиҰҒж”№еҶҷй»ҳи®Өе®һзҺ°гҖӮ

public class Menu extends MenuComponent {
    private List<MenuComponent> menuComponentList;

    public Menu(String name,int level){
        this.level = level;
        this.name = name;
        menuComponentList = new ArrayList<MenuComponent>();
    }

    @Override
    public void add(MenuComponent menuComponent) {
        menuComponentList.add(menuComponent);
    }

    @Override
    public void remove(MenuComponent menuComponent) {
        menuComponentList.remove(menuComponent);
    }

    @Override
    public MenuComponent getChild(int i) {
        return menuComponentList.get(i);
    }

    @Override
    public void print() {

        for (int i = 1; i < level; i++) {
            System.out.print("--");
        }
        System.out.println(name);
        for (MenuComponent menuComponent : menuComponentList) {
            menuComponent.print();
        }
    }
}

Menuзұ»е·Із»Ҹе®һзҺ°дәҶйҷӨдәҶgetNameж–№жі•зҡ„е…¶д»–жүҖжңүж–№жі•пјҢеӣ дёәMenuзұ»е…·жңүж·»еҠ иҸңеҚ•пјҢз§»йҷӨиҸңеҚ•е’ҢиҺ·еҸ–еӯҗиҸңеҚ•зҡ„еҠҹиғҪгҖӮ

public class MenuItem extends MenuComponent {
    public MenuItem(String name,int level) {
        this.name = name;
        this.level = level;
    }

    @Override
    public void print() {
        for (int i = 1; i < level; i++) {
            System.out.print("--");
        }
        System.out.println(name);
    }
}

MenuItemжҳҜиҸңеҚ•йЎ№пјҢдёҚиғҪеҶҚжңүеӯҗиҸңеҚ•пјҢжүҖд»Ҙж·»еҠ иҸңеҚ•пјҢз§»йҷӨиҸңеҚ•е’ҢиҺ·еҸ–еӯҗиҸңеҚ•зҡ„еҠҹиғҪе№¶дёҚиғҪе®һзҺ°гҖӮ

з»„еҗҲжЁЎејҸзҡ„еҲҶзұ»

еңЁдҪҝз”Ёз»„еҗҲжЁЎејҸж—¶пјҢж №жҚ®жҠҪиұЎжһ„д»¶зұ»зҡ„е®ҡд№үеҪўејҸпјҢжҲ‘д»¬еҸҜе°Ҷз»„еҗҲжЁЎејҸеҲҶдёәйҖҸжҳҺз»„еҗҲжЁЎејҸе’Ңе®үе…Ёз»„еҗҲжЁЎејҸдёӨз§ҚеҪўејҸгҖӮ

  • йҖҸжҳҺз»„еҗҲжЁЎејҸ

    йҖҸжҳҺз»„еҗҲжЁЎејҸдёӯпјҢжҠҪиұЎж №иҠӮзӮ№и§’иүІдёӯеЈ°жҳҺдәҶжүҖжңүз”ЁдәҺз®ЎзҗҶжҲҗе‘ҳеҜ№иұЎзҡ„ж–№жі•пјҢжҜ”еҰӮеңЁзӨәдҫӢдёӯ MenuComponent еЈ°жҳҺдәҶ addгҖҒremove гҖҒgetChild ж–№жі•пјҢиҝҷж ·еҒҡзҡ„еҘҪеӨ„жҳҜзЎ®дҝқжүҖжңүзҡ„жһ„д»¶зұ»йғҪжңүзӣёеҗҢзҡ„жҺҘеҸЈгҖӮйҖҸжҳҺз»„еҗҲжЁЎејҸд№ҹжҳҜз»„еҗҲжЁЎејҸзҡ„ж ҮеҮҶеҪўејҸгҖӮ

    йҖҸжҳҺз»„еҗҲжЁЎејҸзҡ„зјәзӮ№жҳҜдёҚеӨҹе®үе…ЁпјҢеӣ дёәеҸ¶еӯҗеҜ№иұЎе’Ңе®№еҷЁеҜ№иұЎеңЁжң¬иҙЁдёҠжҳҜжңүеҢәеҲ«зҡ„пјҢеҸ¶еӯҗеҜ№иұЎдёҚеҸҜиғҪжңүдёӢдёҖдёӘеұӮж¬Ўзҡ„еҜ№иұЎпјҢеҚідёҚеҸҜиғҪеҢ…еҗ«жҲҗе‘ҳеҜ№иұЎпјҢеӣ жӯӨдёәе…¶жҸҗдҫӣ add()гҖҒremove() зӯүж–№жі•жҳҜжІЎжңүж„Ҹд№үзҡ„пјҢиҝҷеңЁзј–иҜ‘йҳ¶ж®өдёҚдјҡеҮәй”ҷпјҢдҪҶеңЁиҝҗиЎҢйҳ¶ж®өеҰӮжһңи°ғз”Ёиҝҷдәӣж–№жі•еҸҜиғҪдјҡеҮәй”ҷпјҲеҰӮжһңжІЎжңүжҸҗдҫӣзӣёеә”зҡ„й”ҷиҜҜеӨ„зҗҶд»Јз Ғпјү

  • е®үе…Ёз»„еҗҲжЁЎејҸ

    еңЁе®үе…Ёз»„еҗҲжЁЎејҸдёӯпјҢеңЁжҠҪиұЎжһ„д»¶и§’иүІдёӯжІЎжңүеЈ°жҳҺд»»дҪ•з”ЁдәҺз®ЎзҗҶжҲҗе‘ҳеҜ№иұЎзҡ„ж–№жі•пјҢиҖҢжҳҜеңЁж ‘жһқиҠӮзӮ№ Menu зұ»дёӯеЈ°жҳҺе№¶е®һзҺ°иҝҷдәӣж–№жі•гҖӮе®үе…Ёз»„еҗҲжЁЎејҸзҡ„зјәзӮ№жҳҜдёҚеӨҹйҖҸжҳҺпјҢеӣ дёәеҸ¶еӯҗжһ„д»¶е’Ңе®№еҷЁжһ„д»¶е…·жңүдёҚеҗҢзҡ„ж–№жі•пјҢдё”е®№еҷЁжһ„д»¶дёӯйӮЈдәӣз”ЁдәҺз®ЎзҗҶжҲҗе‘ҳеҜ№иұЎзҡ„ж–№жі•жІЎжңүеңЁжҠҪиұЎжһ„д»¶зұ»дёӯе®ҡд№үпјҢеӣ жӯӨе®ўжҲ·з«ҜдёҚиғҪе®Ңе…Ёй’ҲеҜ№жҠҪиұЎзј–зЁӢпјҢеҝ…йЎ»жңүеҢәеҲ«ең°еҜ№еҫ…еҸ¶еӯҗжһ„д»¶е’Ңе®№еҷЁжһ„д»¶гҖӮ

дјҳзӮ№

  • з»„еҗҲжЁЎејҸеҸҜд»Ҙжё…жҘҡең°е®ҡд№үеҲҶеұӮж¬Ўзҡ„еӨҚжқӮеҜ№иұЎпјҢиЎЁзӨәеҜ№иұЎзҡ„е…ЁйғЁжҲ–йғЁеҲҶеұӮж¬ЎпјҢе®ғи®©е®ўжҲ·з«ҜеҝҪз•ҘдәҶеұӮж¬Ўзҡ„е·®ејӮпјҢж–№дҫҝеҜ№ж•ҙдёӘеұӮж¬Ўз»“жһ„иҝӣиЎҢжҺ§еҲ¶гҖӮ
  • е®ўжҲ·з«ҜеҸҜд»ҘдёҖиҮҙең°дҪҝз”ЁдёҖдёӘз»„еҗҲз»“жһ„жҲ–е…¶дёӯеҚ•дёӘеҜ№иұЎпјҢдёҚеҝ…е…іеҝғеӨ„зҗҶзҡ„жҳҜеҚ•дёӘеҜ№иұЎиҝҳжҳҜж•ҙдёӘз»„еҗҲз»“жһ„пјҢз®ҖеҢ–дәҶе®ўжҲ·з«Ҝд»Јз ҒгҖӮ
  • еңЁз»„еҗҲжЁЎејҸдёӯеўһеҠ ж–°зҡ„ж ‘жһқиҠӮзӮ№е’ҢеҸ¶еӯҗиҠӮзӮ№йғҪеҫҲж–№дҫҝпјҢж— йЎ»еҜ№зҺ°жңүзұ»еә“иҝӣиЎҢд»»дҪ•дҝ®ж”№пјҢз¬ҰеҗҲвҖңејҖй—ӯеҺҹеҲҷвҖқгҖӮ
  • з»„еҗҲжЁЎејҸдёәж ‘еҪўз»“жһ„зҡ„йқўеҗ‘еҜ№иұЎе®һзҺ°жҸҗдҫӣдәҶдёҖз§ҚзҒөжҙ»зҡ„и§ЈеҶіж–№жЎҲпјҢйҖҡиҝҮеҸ¶еӯҗиҠӮзӮ№е’Ңж ‘жһқиҠӮзӮ№зҡ„йҖ’еҪ’з»„еҗҲпјҢеҸҜд»ҘеҪўжҲҗеӨҚжқӮзҡ„ж ‘еҪўз»“жһ„пјҢдҪҶеҜ№ж ‘еҪўз»“жһ„зҡ„жҺ§еҲ¶еҚҙйқһеёёз®ҖеҚ•гҖӮ

дҪҝз”ЁеңәжҷҜ

з»„еҗҲжЁЎејҸжӯЈжҳҜеә”ж ‘еҪўз»“жһ„иҖҢз”ҹпјҢжүҖд»Ҙз»„еҗҲжЁЎејҸзҡ„дҪҝз”ЁеңәжҷҜе°ұжҳҜеҮәзҺ°ж ‘еҪўз»“жһ„зҡ„ең°ж–№гҖӮжҜ”еҰӮпјҡж–Үд»¶зӣ®еҪ•жҳҫзӨәпјҢеӨҡзә§зӣ®еҪ•е‘ҲзҺ°зӯүж ‘еҪўз»“жһ„ж•°жҚ®зҡ„ж“ҚдҪңгҖӮ

иҪ¬иҪҪиҜ·жіЁжҳҺжқҘжәҗпјҢж¬ўиҝҺеҜ№ж–Үз« дёӯзҡ„еј•з”ЁжқҘжәҗиҝӣиЎҢиҖғиҜҒпјҢж¬ўиҝҺжҢҮеҮәд»»дҪ•жңүй”ҷиҜҜжҲ–дёҚеӨҹжё…жҷ°зҡ„иЎЁиҫҫгҖӮеҸҜд»ҘеңЁдёӢйқўиҜ„и®әеҢәиҜ„и®әпјҢд№ҹеҸҜд»ҘйӮ®д»¶иҮі 1205362172@qq.com
